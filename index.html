async function main() { const loadingText = document.getElementById('loading'); try { // 1. Pyodideのロード let pyodide = await loadPyodide(); // 2. パッケージの準備 await pyodide.loadPackage(["micropip"]); const micropip = pyodide.pyimport("micropip"); loadingText.innerText = "依存ライブラリをインストール中..."; // ここでrequestsをインストールするように指定します await micropip.install("requests"); loadingText.innerText = "ファイルを読み込み中..."; // 3. ファイルを仮想環境内に取り込む const filesToLoad = ['main.py', 'main.kv']; for (const file of filesToLoad) { const response = await fetch(file); if (!response.ok) throw new Error(`${file} が見つかりません`); const content = await response.text(); pyodide.FS.writeFile(file, content); } loadingText.innerText = "アプリを開始します..."; // 4. アプリ実行 const mainCode = pyodide.FS.readFile('main.py', { encoding: 'utf8' }); await pyodide.runPythonAsync(mainCode); loadingText.style.display = 'none'; } catch (err) { console.error(err); loadingText.innerText = "エラー: " + err.message; } }
